#include <cstdio>
#include <iostream>
using namespace std;
typedef long long ll;
ll n;
int main(){
    while(cin >> n){
        cout << 1ll * n * (n - 1) / 2 - 1ll * (n / 2) * (n - n / 2) << '\n';
    }
    return 0;
}
